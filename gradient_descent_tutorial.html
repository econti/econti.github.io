def residual_sum_of_squares(df, frontage_coefficient, total_room_coefficient, y_int):
    rss = 0
    for index, row in df.iterrows():
        prediction = frontage_coefficient * row['LotFrontage'] + \
                     total_room_coefficient * row['TotRmsAbvGrd'] + \
                     y_int
        rss += (prediction - row['SalePrice']) ** 2
    return rss

def step(b0_current, b1_current, b2_current, df, learning_rate):
    """
    b0 = intercept
    b1 = LotFrontage
    b2 = TotRmsAbvGrd
    """
    b0_gradient, b1_gradient, b2_gradient = 0, 0, 0

    for index, row in df.iterrows():
        y, x1, x2 = row['SalePrice'], row['LotFrontage'], row['TotRmsAbvGrd']
        b0_gradient += -2 * (y - (b0_current + (b1_current * x1) + (b2_current * x2)))
        b1_gradient += -2 * x1 * (y - (b0_current + (b1_current * x1) + (b2_current * x2)))
        b2_gradient += -2 * x2 * (y - (b0_current + (b1_current * x1) + (b2_current * x2)))
    
    b0 = b0_current - (learning_rate * b0_gradient)
    b1 = b1_current - (learning_rate * b1_gradient)
    b2 = b2_current - (learning_rate * b2_gradient)
    
    return b0, b1, b2
